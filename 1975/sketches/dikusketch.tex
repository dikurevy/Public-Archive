\begin{roles}

\role{CH}[Anker Helms Jørgensen] Chauffør

\role{K}[Erik Meiling] Kandidat
\role{MS}[Karsten Kynde] Maskinstormer
\end{roles}

\begin{sketch}

  \scene{Lissi sidder ved sit skrivebord, arbejende. Lyd:
    Billarm+sammenstød}

  \says{L} Hvad var det?

  \scene{En chauffør kommer ind.}

  \says{CH} Det må de meget undskylde.  Jeg kom vist til at køre min
  lastbil ind i deres kontor.

  \says{L} En lastbil?

  \says{CH} Ja, last ind, først ud, som man siger.  Jeg kom netop med
  en ladning bits til RECKUs nye bygning, da der pludselig bakker en
  monitor ud foran mig, og jeg må dreje alt det styresystemet kan
  klare for ikke at gå på den store lagringsstak udenfor.

  \says{L} Har De noget imod at køre med den igen -- jeg har lidt
  ravlt; jeg skal have sendt en brevbombe til Morten Lange.

  \says{CH} Jo, javel.  Må jeg rykke bordet lidt.

  \says{L}[hjælper med bordet] Gerne -- hvorfor?

  \says{CH} Så har jeg plads til at vende, forstår De. \act{kører ud}
  \act{En mand ind på scenen.  Han bærer hjelm og et skilt: "`adgang
    forbudt"'. Han er sikkerhedsrepræsentant}

  \says{SR} Hør De der -- De bærer ikke hjelm.

  \says{L} Hjelm?

  \says{SR} Ser De ikke, det er farligt at arbejde her.  Jeg er
  sikkerhedsrepræsentant her.  De bør bære hjelm under arbejdet.

  \says{L} Jamen jeg arbejder ikke.  Forstår De, det er et
  universitetsinstitut, og...

  \says{SR} Åh, jeg forstår.  Jeg havde ellers forstået, der skulle
  være en mase arbejder her.

  \says{L} Arbejder?  Nåh, jamen det er jo studenternes \act{lukker et
    skab op} Vi KALDER dem arbejder.

  \says{SR} Sikke en masse papir.

  \says{L} Ja, det er jo det, vi kan.

  \says{SR} De burde have en sekretær til alt det.

  \says{L} Vi har prøvet, men de løber lige så hurtigt igen.  Derfor
  er vi nødt til at lukke døren så tidligt.  Ellers løber de bare ud.
  Det mindre mig om, at vi må have lukket det hul, lastvognen lavede i
  muren -- her bliver et forfærdeligt TAP-gennemtræk.

  \says{K}[ind. Til SR] Goddag hr. professor, jeg...

  \says{SR} Øh, jeg er egentlig ikke "=

  \says{L} Professoren passer kantinen -- kan jeg hjælpe dem?

  \says{K} ProFESsoren?!

  \says{L} Den ene.  Noget skal de jo bestille.  Den anden er hjemme.

  \says{K} Er han dårlig?

  \says{L} M-j-nåæh"=

  \says{K} Nej, jeg mener, er han syg?

  \says{L} Nåh, næ, han simulerer.

  \says{K} Jeg ville høre om jeg kunne få stilling her?

  \says{L} Næ, vores stillinger ligger i skuffen.

  \says{K} Det var en skuffelse.

  \says{SR} Hi, hi, den forstod jeg!

  \says{L} Hvad kan de?

  \says{K} Kan?  Jeg?  Altså -- jeg er {\em kandidat}.

  \says{L} I datalogi?

  \says{K} Ja.

  \says{L} Det var en skam.  Sådan en har vi.

  \says{K} Men jeg -- det \act{exalteret, griber Lissi i armen} Jeg er
  VIDENSKABSMAND!  Jeg har visioner.  Jeg vil vandre i universitetets
  hellige bibliotekers haller"=

  \says{L} Haller?

  \says{K} "= søgende og FORSKENDE efter appelsinen i min turban; efter Alladins lampe --

  \says{L} Det må være Dat2.

  \says{MS}[ind] Forskning for folket -- ikke for profitten!

  \says{K} Hvem er De, min gode mand?

  \says{MS}[løfter en knyttet hånd] Rød front, og al det der.

  \says{SR}[blander sig i samtalen] Vi må have adgangsbegrænsning; skal vi nyttige borgere betale for sådanne--

  \says{MS} Vi må have flere instruktorer.

  \says{Medlemmer af FLAB om bordet} Flere rejsepenge!

  Har du wåt i Århus?

  Flere bøger på biblioteket!

  Men færre store opgaver.

  Og basser til formiddagskaffen.

  \scene{Jubel - almindelig larm.  Herunder kommer chaufføren ind}

\says{L} Hør du, med lastvognen -- hvad om vi kørte en tur i skoven?

\says{CH} Mægtigt!  Jeg er alligevel netop blevet fyret på RECKU "= \act{Begge ud til venstre}

\scene{TÆPPE}

\end{sketch}
